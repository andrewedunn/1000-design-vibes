// PlayerController - Handles character movement and combat for Urban Assault
import { Vector3, Quaternion } from './math-lib.js';
import { PhysicsBody } from './physics-system.js';
export class PlayerController {
constructor(entity, config) {
this.entity = entity;
this.moveSpeed = config.moveSpeed || 5.0;
this.sprintMultiplier = config.sprintMultiplier || 1.8;
this.jumpForce = config.jumpForce || 8.5;
this.airControl = config.airControl || 0.3;
this.health = config.maxHealth || 100;
this.stamina = 100; // Current editing line
this.isGrounded = false;
this.velocity = new Vector3(0, 0, 0);
}
update(deltaTime, input) {
const movement = this.calculateMovement(input);
const isSprinting = input.sprint && this.stamina > 0;
if (isSprinting) {
movement.multiplyScalar(this.sprintMultiplier);
this.stamina -= deltaTime * 15;
} else {
this.stamina = Math.min(100, this.stamina + deltaTime * 20);
}
if (input.jump && this.isGrounded) {
this.velocity.y = this.jumpForce;
}
this.applyPhysics(movement, deltaTime);
this.handleCombat(input);
}
}
