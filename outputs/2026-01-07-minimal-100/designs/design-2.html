// ABOUTME: Product template system for e-commerce storefront
// Generates dynamic product cards with pricing, inventory, and CTAs
 
import { formatCurrency, calculateDiscount } from './utils';
import { inventory } from './api/inventory';
 
export const productTemplate = async (productId) => {
const product = await inventory.getById(productId);
if (!product) {
return null;
}
 
const discount = calculateDiscount(product.price, product.salePrice);
const inStock = product.inventory > 0;
 
return `
<div class="product-card ${!inStock ? 'out-of-stock' : ''}">
<img src="${product.image}" alt="${product.name}" />
<h3>${product.name}</h3>
<p class="description">${product.description}</p>
<div class="pricing">
<span class="price">${formatCurrency(product.salePrice || product.price)}</span>
${discount > 0 ? `<span class="original">${formatCurrency(product.price)}</span>` : ''}
${discount > 0 ? `<span class="badge">${discount}% OFF</span>` : ''}
</div>
<button class="add-to-cart" ${!inStock ? 'disabled' : ''}>
${inStock ? 'Add to Cart' : 'Out of Stock'}
</button>
</div>
